```brainfuck

>++++++++++[-<++++++++++>]
( N=100 'n=0 )
<[->+
  [->+>+>>>+>>+<<<<<<<]
  ( N '0 n n 0 0 n 0 n )
  >[-<+>]
  >>+
  ( N n 0 n '1 0 n 0 n )
  n模3
  >>>>[
    ->+<[ ->+<[
      ->[-]
    ]]
  <]<[>]
  ( N n 0 n 1 0 n '0 0 {n%3} )

  >+
  >[[-]<->]
  ( N n 0 n 1 0 n 0 {n%3==0?1:0} '0 )
  <[[-]
    <<<<<[-]
    >>>>>>+++++++[-<++++++++++>]<. F
    >+++++[-<+++++++>]<. i
    >++++[-<++++>]<+.. zz
    [-]
  ]
  ( N n 0 {n%3==0?0:n} 1 0 n 0 '0 )
```
