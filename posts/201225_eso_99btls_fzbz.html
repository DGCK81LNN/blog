初始化计数器
>++++++++++[-<++++++++++>]
( N=100 'n=0 )
<[->+
  [->+>+>>>+>>+<<<<<<<]
  ( N '0 n n 0 0 n 0 n )
  >[-<+>]
  >>+
  ( N n 0 n '1 0 n 0 n )
  n模3
  >>>>[
    ->+<[ ->+<[
      ->[-]
    ]]
  <]<[>]
  ( N n 0 n 1 0 n '0 0 {n%3} )
  如果余数为0就输出Fizz
  >+
  >[[-]<->]
  ( N n 0 n 1 0 n 0 {n%3==0?1:0} '0 )
  <[[-]
    <<<<<[-]
    >>>>>>+++++++[-<++++++++++>]<. F
    >+++++[-<+++++++>]<. i
    >++++[-<++++>]<+.. zz
    [-]
  ]
  ( N n 0 {n%3==0?0:n} 1 0 n 0 '0 )
  n模5
  <<[
    ->+<[ ->+<[ ->+<[ ->+<[
      ->[-]
    ]]]]
  <]<[>]
  ( N n 0 {n%3==0?0:n} 1 '0 0 {n%5} )
  如果余数为0就输出Buzz
  >+
  >[[-]<->]
  ( N n 0 {n%3==0?0:n} 1 0 {n%5==0} '0 )
  <[[-]
    <<<[-]
    >>>>++++++[-<+++++++++++>]<. B
    >+++++++[-<+++++++>]<++. u
    +++++.. zz
    [-]
  ]
